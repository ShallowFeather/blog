    var S = [], j = 0, temp, plaintext = '';
    for (var i = 0; i < 256; i++) {
        S[i] = i;

    for (i = 0; i < 256; i++) {

        temp = S[i];
        S[i] = S[j];
        S[j] = temp;

    i = 0;

    for (var k = 0; k < cipherText.length; k++) {
